    for element := this.Front(); element != nil; element = element.Next() {

        this.PushFront(value)
        return
    }
    if this.List.Len() < this.Limit && this.compareFunc(value, this.Back().Value) {
        this.PushBack(value)
        return

    }

        this.Remove(this.Back())
    }
